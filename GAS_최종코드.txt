// 검토중 상태 전원에게 리마인드 일괄 발송
function sendBulkReminder() {
  const ui    = SpreadsheetApp.getUi();
  const sheet = SpreadsheetApp.getActiveSpreadsheet().getSheets()[0];
  const data  = sheet.getDataRange().getValues();

  let count = 0;
  for (let i = 1; i < data.length; i++) {
    const status = String(data[i][10] || '').trim();
    if (status !== '검토중') continue;
    if (!data[i][2] || !data[i][6]) continue;
    sendReminderSMS(sheet, i + 1);
    count++;
    Utilities.sleep(300);
  }

  ui.alert(`리마인드 발송 완료: ${count}명에게 발송되었습니다.`);
}

function doPost(e) {
  try {
    // 뱅크다 요청: query param ?type= 으로 구분
    const reqType = e.parameter && e.parameter.type;
    if (reqType === 'orders')  return handleBankdaOrders();
    if (reqType === 'detail')  return handleBankdaDetail(e);
    if (reqType === 'confirm') return handleBankdaConfirm(e);

    // 일반 신청 처리
    const data = JSON.parse(e.postData.contents);
    if (data.type === 'crew')     return handleCrew(data);
    if (data.type === 'guest')    return handleGuest(data);
    if (data.type === 'adReport') return handleAdReport(data);
    return handleApply(data);

  } catch (err) {
    return jsonResponse({ result: 'error', message: err.message });
  }
}

// 한국어 날짜 문자열 → "yyyy-MM-dd HH:mm:ss" 변환
function parseOrderDate(val) {
  if (val instanceof Date) {
    return Utilities.formatDate(val, 'Asia/Seoul', 'yyyy-MM-dd HH:mm:ss');
  }
  var s = String(val);
  var m = s.match(/(\d{4})\.\s*(\d{1,2})\.\s*(\d{1,2})\.\s*(오전|오후)\s*(\d{1,2}):(\d{2}):(\d{2})/);
  if (!m) return s;
  var hour = parseInt(m[5]);
  if (m[4] === '오후' && hour !== 12) hour += 12;
  if (m[4] === '오전' && hour === 12) hour = 0;
  return m[1] + '-' + m[2].padStart(2,'0') + '-' + m[3].padStart(2,'0') + ' '
       + String(hour).padStart(2,'0') + ':' + m[6] + ':' + m[7];
}

// 뱅크다 → 미확인 주문 리스트 요청
function handleBankdaOrders() {
  const sheet      = SpreadsheetApp.getActiveSpreadsheet().getSheets()[0];
  const data       = sheet.getDataRange().getValues();
  const orders     = [];
  const unpaidSet  = ['검토중', '리마인드✓발송'];

  for (let i = 1; i < data.length; i++) {
    const row    = data[i];
    const status = String(row[10] || '').trim();
    if (unpaidSet.indexOf(status) === -1) continue;

    const name   = String(row[2] || '');
    const ticket = String(row[7] || '');
    const price  = parseInt(String(row[8] || '').replace(/[^0-9]/g, '')) || 0;
    if (!name || !price) continue;

    orders.push({
      order_id:           'row_' + (i + 1),
      buyer_name:         name,
      billing_name:       name,
      bank_account_no:    OUR_BANK_ACCOUNT,
      bank_code_name:     OUR_BANK_NAME,
      order_price_amount: price,
      order_date:         parseOrderDate(row[0]),
      items: [{ product_name: ticket }]
    });
  }

  return jsonResponse({ orders });
}

// 뱅크다 → 주문 상세 요청
function handleBankdaDetail(e) {
  const body   = JSON.parse(e.postData.contents);
  const rowNum = parseInt(String(body.order_id || '').replace('row_', ''));
  if (!rowNum) return jsonResponse({ result: 'error', message: 'invalid order_id' });

  const sheet = SpreadsheetApp.getActiveSpreadsheet().getSheets()[0];
  const row   = sheet.getRange(rowNum, 1, 1, 11).getValues()[0];
  const price = parseInt(String(row[8] || '').replace(/[^0-9]/g, '')) || 0;

  return jsonResponse({
    order: {
      order_id:           'row_' + rowNum,
      buyer_name:         String(row[2] || ''),
      billing_name:       String(row[2] || ''),
      bank_account_no:    OUR_BANK_ACCOUNT,
      bank_code_name:     OUR_BANK_NAME,
      order_price_amount: price,
      order_date:         parseOrderDate(row[0]),
      items: [{ product_name: String(row[7] || '') }]
    }
  });
}

// 뱅크다 → 입금 매칭 완료 통보
function handleBankdaConfirm(e) {
  const body     = JSON.parse((e.postData || {}).contents || '{}');
  const requests = body.requests || [];
  const sheet    = SpreadsheetApp.getActiveSpreadsheet().getSheets()[0];
  const unpaidSet = ['검토중', '리마인드✓발송'];
  const results  = [];

  for (var i = 0; i < requests.length; i++) {
    var orderId = requests[i].order_id || '';
    var rowNum  = parseInt(String(orderId).replace('row_', ''));
    if (!rowNum) {
      results.push({ order_id: orderId, description: '존재하지 않는 주문' });
      continue;
    }

    var status = String(sheet.getRange(rowNum, 11).getValue()).trim();
    if (unpaidSet.indexOf(status) === -1) {
      results.push({ order_id: orderId, description: '요청된 주문번호가 입금대기 상태가 아닌 경우' });
      continue;
    }

    sheet.getRange(rowNum, 11).setValue('입금완료');
    sheet.getRange(rowNum, 1, 1, 14).setBackground('#e8f5e9');

    // 입금 확인 = 실제 결제 완료 → Purchase CAPI 전송
    var rowData  = sheet.getRange(rowNum, 1, 1, 14).getValues()[0];
    var priceNum = parseInt(String(rowData[8] || '').replace(/[^0-9]/g, '')) || 0;
    sendMetaCAPIPurchase({
      phone:  String(rowData[6] || ''),
      name:   String(rowData[2] || ''),
      gender: String(rowData[3] || ''),
      price:  priceNum,
      dob:    parseDobFromSheet(String(rowData[4] || '')),
      fbp:    String(rowData[11] || ''),
      fbc:    String(rowData[12] || ''),
      ua:     String(rowData[13] || '')
    });

    results.push({ order_id: orderId, description: '성공' });
  }

  return jsonResponse({ return_code: 200, description: '정상', orders: results });
}

function onEdit(e) {
  try {
    if (!e || !e.range) return;
    const sheet     = e.range.getSheet();
    const sheetName = sheet.getName();
    if (sheetName === '크루지원') return;

    const col   = e.range.getColumn();
    const row   = e.range.getRow();
    if (row < 2) return;

    const value = (e.value || '').toString().trim();

    // 게스트명단: H열(8번) "입금완료"
    if (sheetName === '게스트명단') {
      if (col !== 8 || value !== '입금완료') return;
      sendGuestConfirmSMS(sheet, row);
      addGuestToRoster(sheet, row);
      return;
    }

    // 파티신청: K열(11번)
    if (col !== 11) return;
    if      (value === '합격')    { sendApprovalSMS(sheet, row); addToCurrentRoster(sheet, row); }
    else if (value === '리마인드') { sendReminderSMS(sheet, row); }
    else if (value === '불합격')  { sendRejectionSMS(sheet, row); }
    else if (value === '취소')    { removeFromCurrentRoster(sheet, row); }

  } catch (err) {
    Logger.log('[onEdit 오류] ' + err.message + ' | 시트: ' + (e && e.range ? e.range.getSheet().getName() : '?') + ' | 셀: ' + (e && e.range ? e.range.getA1Notation() : '?') + ' | 값: ' + (e && e.value || ''));
  }
}

function sendSMS(to, msg) {
  try {
    let cleanPhone = to.toString().replace(/[^0-9]/g, '');
    if (cleanPhone.length === 10 && cleanPhone.startsWith('10')) {
      cleanPhone = '0' + cleanPhone;
    }
    const date      = new Date();
    const salt      = Utilities.getUuid().replace(/-/g, '');
    const dateStr   = Utilities.formatDate(date, 'UTC', "yyyy-MM-dd'T'HH:mm:ss'Z'");
    const signature = computeHmacSha256(SOLAPI_SECRET, dateStr + salt);

    UrlFetchApp.fetch('https://api.solapi.com/messages/v4/send', {
      method: 'post',
      contentType: 'application/json',
      headers: {
        'Authorization': `HMAC-SHA256 apiKey=${SOLAPI_KEY}, date=${dateStr}, salt=${salt}, signature=${signature}`
      },
      payload: JSON.stringify({ message: { to: cleanPhone, from: SENDER, text: msg } }),
      muteHttpExceptions: true
    });
  } catch (err) {
    // 발송 실패해도 신청 자체는 정상 처리
  }
}

function computeHmacSha256(secret, message) {
  const rawSig = Utilities.computeHmacSha256Signature(
    Utilities.newBlob(message).getBytes(),
    Utilities.newBlob(secret).getBytes()
  );
  return rawSig.map(b => ('0' + (b & 0xFF).toString(16)).slice(-2)).join('');
}

function sendMetaCAPI(data) {
  try {
    function sha256(value) {
      if (!value) return null;
      const bytes = Utilities.computeDigest(
        Utilities.DigestAlgorithm.SHA_256,
        value.toString().trim().toLowerCase()
      );
      return bytes.map(b => ('0' + (b & 0xFF).toString(16)).slice(-2)).join('');
    }

    function normalizePhone(phone) {
      let p = phone.replace(/\D/g, '');
      if (p.startsWith('0')) p = '82' + p.slice(1);
      return p;
    }

    const genderCode = data.gender === '여성' ? 'f' : 'm';
    const dob = data.byear
      ? data.byear + String(data.bmonth || '').padStart(2,'0') + String(data.bday || '').padStart(2,'0')
      : null;

    const payload = {
      data: [{
        event_name: 'Lead',
        event_time: Math.floor(new Date().getTime() / 1000),
        event_id: data.eventId || ('gas_' + Date.now()),
        action_source: 'website',
        event_source_url: 'https://sumitparty.com/apply.html?date=' + (data.date || ''),
        user_data: {
          ph:  data.phone ? sha256(normalizePhone(data.phone)) : null,
          fn:  (data.name && data.name.length > 1) ? sha256(data.name.slice(1)) : null,
          ln:  data.name  ? sha256(data.name.slice(0, 1))      : null,
          ge:  sha256(genderCode),
          db:  dob ? sha256(dob) : null,
          client_ip_address: data.clientIp  || null,
          client_user_agent: data.userAgent || null,
          fbc: data.fbc || null,
          fbp: data.fbp || null
        }
      }]
    };

    // null 필드 제거
    const ud = payload.data[0].user_data;
    Object.keys(ud).forEach(k => { if (ud[k] === null) delete ud[k]; });

    const url = 'https://graph.facebook.com/v21.0/' + PIXEL_ID + '/events?access_token=' + CAPI_ACCESS_TOKEN;

    UrlFetchApp.fetch(url, {
      method: 'POST',
      contentType: 'application/json',
      payload: JSON.stringify(payload),
      muteHttpExceptions: true
    });
  } catch(err) {
    // 조용히 실패 — 폼 제출은 막지 않음
  }
}

function generateWeeklyRosterForDate(partyDate) {
  if (!partyDate || isNaN(new Date(partyDate))) return;

function generateWeeklyRosterForDates(dates) {
  var ss       = SpreadsheetApp.getActiveSpreadsheet();
  var dayNames = ['일', '월', '화', '수', '목', '금', '토'];
  var BUFFER   = 5;

  var friday     = dates[0];
  var rosterName = '명단_' + (friday.getMonth()+1) + '월' + friday.getDate() + '일주';
  var rosterSheet = ss.getSheetByName(rosterName);
  if (!rosterSheet) {
    rosterSheet = ss.insertSheet(rosterName);
  } else {
    rosterSheet.clearContents();
    rosterSheet.clearFormats();
  }

  var mainSheet  = ss.getSheets()[0];
  var mainData   = mainSheet.getDataRange().getValues();
  var guestSheet = ss.getSheetByName('게스트명단');
  var guestData  = guestSheet ? guestSheet.getDataRange().getValues() : [];

  var currentRow = 1;

  dates.forEach(function(date) {
    var dateStr   = Utilities.formatDate(date, 'Asia/Seoul', 'yyyy-MM-dd');
    var dow       = date.getDay();
    var dateLabel = (date.getMonth() + 1) + '월 ' + date.getDate() + '일 (' + dayNames[dow] + ')';

    var maleList = [], femaleList = [];

    for (var i = 1; i < mainData.length; i++) {
      var row = mainData[i];
      var rowDateStr = row[1] instanceof Date
        ? Utilities.formatDate(row[1], 'Asia/Seoul', 'yyyy-MM-dd')
        : String(row[1]).substring(0, 10);
      var status = String(row[10] || '').trim();
      if (rowDateStr !== dateStr || status.indexOf('합격') === -1) continue;
      if (String(row[3]).indexOf('남') !== -1) {
        maleList.push([row[2], row[6], row[7]]);
      } else {
        femaleList.push([row[2], row[6], row[7]]);
      }
    }

    for (var j = 1; j < guestData.length; j++) {
      var grow = guestData[j];
      var growDateStr = grow[1] instanceof Date
        ? Utilities.formatDate(grow[1], 'Asia/Seoul', 'yyyy-MM-dd')
        : String(grow[1]).substring(0, 10);
      var gstatus = String(grow[7] || '').trim();
      if (growDateStr !== dateStr || gstatus.indexOf('입금완료') === -1) continue;
      var ggender = String(grow[3] || '').trim();
      if (ggender.indexOf('남') !== -1) {
        maleList.push([grow[2], grow[4], '게스트(' + grow[5] + ')']);
      } else {
        femaleList.push([grow[2], grow[4], '게스트(' + grow[5] + ')']);
      }
    }

    var ROWS = Math.max(maleList.length, femaleList.length, 50) + BUFFER;

    rosterSheet.getRange(currentRow, 1, 1, 7).merge()
      .setValue('📅 ' + dateLabel)
      .setBackground('#1a1a1a').setFontColor('#C9A96E')
      .setFontWeight('bold').setFontSize(13)
      .setHorizontalAlignment('center');
    currentRow++;

    var dataStart = currentRow + 2;
    var dataEnd   = dataStart + ROWS - 1;
    rosterSheet.getRange(currentRow, 1, 1, 3).merge()
      .setFormula('="남성 — "&COUNTA(A'+dataStart+':A'+dataEnd+')&"명 / 50명"')
      .setBackground('#0d1020').setFontColor('#8899ff').setFontWeight('bold');
    rosterSheet.getRange(currentRow, 5, 1, 3).merge()
      .setFormula('="여성 — "&COUNTA(E'+dataStart+':E'+dataEnd+')&"명 / 50명"')
      .setBackground('#200d0d').setFontColor('#ff8888').setFontWeight('bold');
    currentRow++;

    rosterSheet.getRange(currentRow, 1, 1, 3).setValues([['성함', '전화번호', '권종']])
      .setFontWeight('bold').setBackground('#2a2a2a').setFontColor('#ffffff');
    rosterSheet.getRange(currentRow, 5, 1, 3).setValues([['성함', '전화번호', '권종']])
      .setFontWeight('bold').setBackground('#2a2a2a').setFontColor('#ffffff');
    currentRow++;

    var maleGrid = [], femaleGrid = [];
    for (var m = 0; m < ROWS; m++) {
      maleGrid.push(m < maleList.length ? maleList[m] : ['', '', '']);
      femaleGrid.push(m < femaleList.length ? femaleList[m] : ['', '', '']);
    }
    rosterSheet.getRange(currentRow, 1, ROWS, 3).setValues(maleGrid);
    rosterSheet.getRange(currentRow, 5, ROWS, 3).setValues(femaleGrid);
    currentRow += ROWS + 2;
  });

  rosterSheet.setColumnWidth(4, 20);
  rosterSheet.autoResizeColumns(1, 3);
  rosterSheet.autoResizeColumns(5, 3);
}

  return getPartyDatesFromBase(new Date());

function generateNextWeekRoster() {
  var next = new Date();

function addToCurrentRoster(mainSheet, rowNum) {
  try {
    var ss = SpreadsheetApp.getActiveSpreadsheet();

    var partyDateVal = mainSheet.getRange(rowNum, 2).getValue();
    var name   = String(mainSheet.getRange(rowNum, 3).getValue() || '').trim();
    var gender = String(mainSheet.getRange(rowNum, 4).getValue() || '').trim();
    var phone  = String(mainSheet.getRange(rowNum, 7).getValue() || '').trim();
    var ticket = String(mainSheet.getRange(rowNum, 8).getValue() || '').trim();
    if (!name || !phone) return;

    var partyDate = partyDateVal instanceof Date ? partyDateVal : new Date(partyDateVal);
    if (isNaN(partyDate.getTime())) return;

    // 이번 주 금요일 기준으로 명단 시트명 계산
    var dates = getPartyDatesFromBase(partyDate);
    var friday = dates[0];
    var rosterName = '명단_' + (friday.getMonth()+1) + '월' + friday.getDate() + '일주';
    var rosterSheet = ss.getSheetByName(rosterName);

    // 명단 시트가 없으면 자동 생성
    if (!rosterSheet) {
      generateWeeklyRosterForDates(dates);
      rosterSheet = ss.getSheetByName(rosterName);
      if (!rosterSheet) return;
    }

    var dateStr = Utilities.formatDate(partyDate, 'Asia/Seoul', 'yyyy-MM-dd');
    var rosterVals = rosterSheet.getDataRange().getValues();

    // 섹션(날짜 블록) 탐지
    var sections = [];
    for (var i = 0; i < rosterVals.length; i++) {
      if (String(rosterVals[i][0] || '').trim().indexOf('📅') === 0) {
        sections.push({ headerRowIdx: i, dataStartRowIdx: i + 3 });
      }
    }

    // 파티 날짜와 일치하는 섹션 찾기 (금/토/일 순서로 매핑)
    var sectionIdx = -1;
    for (var si = 0; si < sections.length && si < dates.length; si++) {
      var sDateStr = Utilities.formatDate(dates[si], 'Asia/Seoul', 'yyyy-MM-dd');
      if (sDateStr === dateStr) { sectionIdx = si; break; }
    }
    if (sectionIdx === -1) return;

    var sec = sections[sectionIdx];
    var dataStart = sec.dataStartRowIdx;
    var dataEnd = (sectionIdx + 1 < sections.length)
      ? sections[sectionIdx + 1].headerRowIdx - 1
      : rosterVals.length - 1;

    // 중복 체크
    var key = name + '|' + phone;
    for (var r = dataStart; r <= dataEnd; r++) {
      var mN = String(rosterVals[r] ? rosterVals[r][0] : '').trim();
      var mP = String(rosterVals[r] ? rosterVals[r][1] : '').trim();
      var fN = String(rosterVals[r] ? rosterVals[r][4] : '').trim();
      var fP = String(rosterVals[r] ? rosterVals[r][5] : '').trim();
      if (mN + '|' + mP === key || fN + '|' + fP === key) return;
    }

    var isMale = gender.indexOf('남') !== -1;
    var colIdx = isMale ? 0 : 4; // 0-indexed: 남=A열(0), 여=E열(4)

    // 빈 슬롯 탐색
    var targetRow = -1;
    for (var r = dataStart; r <= dataEnd; r++) {
      if (!String(rosterVals[r] ? rosterVals[r][colIdx] : '').trim()) {
        targetRow = r; break;
      }
    }

    // 빈 슬롯 없으면 마지막 사용 행 다음에 추가
    if (targetRow === -1) {
      var lastUsed = dataStart;
      for (var r = dataStart; r <= dataEnd; r++) {
        if (String(rosterVals[r] ? rosterVals[r][colIdx] : '').trim()) lastUsed = r;
      }
      targetRow = lastUsed + 1;
    }

    // Sheets는 1-indexed이므로 +1
    rosterSheet.getRange(targetRow + 1, colIdx + 1, 1, 3).setValues([[name, phone, ticket]]);

  } catch(err) {
    Logger.log('[addToCurrentRoster 오류] ' + err.message);
  }
}

// ════════════════════════════════════════════════
//  명단 새로고침 — 현재 열린 명단_ 시트에 누락 인원만 append
// ════════════════════════════════════════════════

function refreshCurrentRoster() {
  var ss          = SpreadsheetApp.getActiveSpreadsheet();
  var rosterSheet = ss.getActiveSheet();
  var sheetName   = rosterSheet.getName();

  if (sheetName.indexOf('명단_') !== 0) {
    SpreadsheetApp.getUi().alert('먼저 명단_ 시트를 열고 실행해주세요.\n현재 시트: ' + sheetName);
    return;
  }

  // 시트 이름에서 기준 날짜 파싱: 명단_6월19일주 → month=6, day=19
  var nm = sheetName.match(/명단_(\d+)월(\d+)일주/);
  if (!nm) {
    SpreadsheetApp.getUi().alert('시트 이름 형식을 인식할 수 없습니다: ' + sheetName);
    return;
  }
  var baseDate = new Date(new Date().getFullYear(), parseInt(nm[1]) - 1, parseInt(nm[2]));
  var dates    = getPartyDatesFromBase(baseDate); // [금, 토, 일]

  // 소스 데이터 읽기
  var mainSheet  = ss.getSheets()[0];
  var mainData   = mainSheet.getDataRange().getValues();
  var guestSheet = ss.getSheetByName('게스트명단');
  var guestData  = guestSheet ? guestSheet.getDataRange().getValues() : [];

  // 현재 명단 전체 읽기
  var rosterVals = rosterSheet.getDataRange().getValues();

  // 섹션 경계 탐지 + 기존 인원 중복 체크 Set 구성
  var sections     = []; // [{ headerRowIdx, dataStartRowIdx }]
  var existingKeys = {};

  for (var i = 0; i < rosterVals.length; i++) {
    var cell0 = String(rosterVals[i][0] || '').trim();

    // 📅 로 시작하는 행 = 날짜 섹션 헤더
    if (cell0.indexOf('📅') === 0) {
      sections.push({ headerRowIdx: i, dataStartRowIdx: i + 3 });
    }

    // 남성 컬럼 (col A=0, col B=1) — 이름 있고 헤더/공식 행이 아닌 경우만
    var mName  = String(rosterVals[i][0] || '').trim();
    var mPhone = String(rosterVals[i][1] || '').trim();
    if (mName && mName !== '성함'
        && mName.indexOf('📅') === -1
        && mName.indexOf('남성') === -1
        && mName.indexOf('여성') === -1) {
      existingKeys[mName + '|' + mPhone] = true;
    }

    // 여성 컬럼 (col E=4, col F=5)
    var fName  = String(rosterVals[i][4] || '').trim();
    var fPhone = String(rosterVals[i][5] || '').trim();
    if (fName && fName !== '성함' && fName.indexOf('여성') === -1) {
      existingKeys[fName + '|' + fPhone] = true;
    }
  }

  var totalAdded = 0;

  dates.forEach(function(date, idx) {
    if (!sections[idx]) return;

    var dateStr   = Utilities.formatDate(date, 'Asia/Seoul', 'yyyy-MM-dd');
    var sec       = sections[idx];
    var dataStart = sec.dataStartRowIdx; // 0-indexed
    var dataEnd   = (idx + 1 < sections.length)
      ? sections[idx + 1].headerRowIdx - 1
      : rosterVals.length - 1;

    // 이 날짜의 신규 인원 찾기
    var newMales = [], newFemales = [];

    for (var i = 1; i < mainData.length; i++) {
      var row      = mainData[i];
      var rDateStr = row[1] instanceof Date
        ? Utilities.formatDate(row[1], 'Asia/Seoul', 'yyyy-MM-dd')
        : String(row[1]).substring(0, 10);
      var status   = String(row[10] || '').trim();
      if (rDateStr !== dateStr || status.indexOf('합격') === -1) continue;

      var name  = String(row[2] || '').trim();
      var phone = String(row[6] || '').trim();
      var key   = name + '|' + phone;
      if (existingKeys[key]) continue;

      if (String(row[3]).indexOf('남') !== -1) {
        newMales.push([row[2], row[6], row[7]]);
      } else {
        newFemales.push([row[2], row[6], row[7]]);
      }
      existingKeys[key] = true;
    }

    for (var j = 1; j < guestData.length; j++) {
      var grow     = guestData[j];
      var gDateStr = grow[1] instanceof Date
        ? Utilities.formatDate(grow[1], 'Asia/Seoul', 'yyyy-MM-dd')
        : String(grow[1]).substring(0, 10);
      var gStatus  = String(grow[7] || '').trim();
      if (gDateStr !== dateStr || gStatus.indexOf('입금완료') === -1) continue;

      var gName  = String(grow[2] || '').trim();
      var gPhone = String(grow[4] || '').trim();
      var gKey   = gName + '|' + gPhone;
      if (existingKeys[gKey]) continue;

      var ggender = String(grow[3] || '').trim();
      if (ggender.indexOf('남') !== -1) {
        newMales.push([grow[2], grow[4], '게스트(' + grow[5] + ')']);
      } else {
        newFemales.push([grow[2], grow[4], '게스트(' + grow[5] + ')']);
      }
      existingKeys[gKey] = true;
    }

    if (newMales.length === 0 && newFemales.length === 0) return;

    // 남성 빈 슬롯 수집 (col A = index 0) + 마지막 사용 행 추적
    var maleSlots = [];
    var lastUsedMale = dataStart - 1;
    for (var r = dataStart; r <= dataEnd; r++) {
      var mCell = String(rosterVals[r] ? rosterVals[r][0] : '').trim();
      if (!mCell) {
        maleSlots.push(r);
      } else {
        lastUsedMale = r;
      }
    }

    // 여성 빈 슬롯 수집 (col E = index 4) + 마지막 사용 행 추적
    var femaleSlots = [];
    var lastUsedFemale = dataStart - 1;
    for (var r = dataStart; r <= dataEnd; r++) {
      var fCell = String(rosterVals[r] ? rosterVals[r][4] : '').trim();
      if (!fCell) {
        femaleSlots.push(r);
      } else {
        lastUsedFemale = r;
      }
    }

    // 남성 쓰기 (col A~C) — 빈 슬롯 소진 시 마지막 행 아래에 자동 추가
    var mSlotIdx = 0;
    for (var mi = 0; mi < newMales.length; mi++) {
      var mTargetRow;
      if (mSlotIdx < maleSlots.length) {
        mTargetRow = maleSlots[mSlotIdx++];
      } else {
        lastUsedMale++;
        mTargetRow = lastUsedMale;
      }
      rosterSheet.getRange(mTargetRow + 1, 1, 1, 3).setValues([newMales[mi]]);
      totalAdded++;
    }

    // 여성 쓰기 (col E~G) — 빈 슬롯 소진 시 마지막 행 아래에 자동 추가
    var fSlotIdx = 0;
    for (var fi = 0; fi < newFemales.length; fi++) {
      var fTargetRow;
      if (fSlotIdx < femaleSlots.length) {
        fTargetRow = femaleSlots[fSlotIdx++];
      } else {
        lastUsedFemale++;
        fTargetRow = lastUsedFemale;
      }
      rosterSheet.getRange(fTargetRow + 1, 5, 1, 3).setValues([newFemales[fi]]);
      totalAdded++;
    }
  });

  SpreadsheetApp.getUi().alert('명단 새로고침 완료! ' + totalAdded + '명이 추가되었습니다.');
}

function getPartyDatesFromBase(base) {
  var today = new Date(base);
  today.setHours(0, 0, 0, 0);
  var dow = today.getDay();

  var daysToFriday;
  if      (dow === 0) { daysToFriday = -2; }
  else if (dow === 6) { daysToFriday = -1; }
  else if (dow === 5) { daysToFriday =  0; }
  else                { daysToFriday = 5 - dow; }

  var friday   = new Date(today); friday.setDate(today.getDate() + daysToFriday);
  var saturday = new Date(friday); saturday.setDate(friday.getDate() + 1);
  var sunday   = new Date(friday); sunday.setDate(friday.getDate() + 2);

  return [friday, saturday, sunday];
}

// ════════════════════════════════════════════════
//  1. 다음 주 파티 인원 집계 (합격 기준)
// ════════════════════════════════════════════════

function getNextWeekPartyCounts() {
  // 일요일 기준 → 다음 주 금요일 계산
  var base = new Date();
  var dow  = base.getDay(); // 0=일
  var daysToNextFriday = (7 - dow + 5) % 7;
  if (daysToNextFriday === 0) daysToNextFriday = 7;
  base.setDate(base.getDate() + daysToNextFriday);

  // getPartyDatesFromBase → [금, 토, 일] 반환, 금/토만 사용
  var dates = getPartyDatesFromBase(base);
  var partyDates = [dates[0], dates[1]]; // 금요일, 토요일

  var ss         = SpreadsheetApp.getActiveSpreadsheet();
  var mainData   = ss.getSheets()[0].getDataRange().getValues();
  var guestSheet = ss.getSheetByName('게스트명단');
  var guestData  = guestSheet ? guestSheet.getDataRange().getValues() : [];
  var dayNames   = ['일','월','화','수','목','금','토'];
  var result     = {};

  partyDates.forEach(function(date) {
    var dateStr = Utilities.formatDate(date, 'Asia/Seoul', 'yyyy-MM-dd');
    var label   = (date.getMonth()+1) + '/' + date.getDate() + '(' + dayNames[date.getDay()] + ')';
    var male = 0, female = 0;

    // 메인 신청 시트: 합격✓발송 포함
    for (var i = 1; i < mainData.length; i++) {
      var r = mainData[i];
      var rDate  = r[1] instanceof Date
        ? Utilities.formatDate(r[1], 'Asia/Seoul', 'yyyy-MM-dd')
        : String(r[1]).substring(0, 10);
      var status = String(r[10] || '').trim();
      if (rDate !== dateStr || status.indexOf('합격') === -1) continue;
      if (String(r[3]).indexOf('남') !== -1) male++;
      else female++;
    }

    // 게스트명단: 입금완료 기준 (성별에 따라 집계)
    for (var j = 1; j < guestData.length; j++) {
      var g = guestData[j];
      var gDate = g[1] instanceof Date
        ? Utilities.formatDate(g[1], 'Asia/Seoul', 'yyyy-MM-dd')
        : String(g[1]).substring(0, 10);
      if (gDate !== dateStr || String(g[7] || '').indexOf('입금완료') === -1) continue;
      if (String(g[3] || '').indexOf('남') !== -1) male++;
      else female++;
    }

    result[dateStr] = { label: label, male: male, female: female };
  });

  return result;
}

//  2. 광고세트 성과 조회 (최근 7일)
// ════════════════════════════════════════════════

function getAdSetPerformance() {
  var adsets = [
    { key: 'FEMALE_MAIN', id: ADSET_FEMALE_MAIN, name: '여성_19-34_부산_잠재고객' },
    { key: 'MALE',        id: ADSET_MALE_MAIN,   name: '남성_부산_잠재고객' }
  ];
  var result = {};

  adsets.forEach(function(as) {
    try {
      var url = 'https://graph.facebook.com/v21.0/' + as.id + '/insights'
        + '?fields=impressions,clicks,ctr,cpm,cpc,spend'
        + '&date_preset=last_7d'
        + '&access_token=' + ADS_ACCESS_TOKEN;
      var res  = UrlFetchApp.fetch(url, { muteHttpExceptions: true });
      var json = JSON.parse(res.getContentText());

      if (json.data && json.data[0]) {
        var d = json.data[0];
        result[as.key] = {
          name:        as.name,
          impressions: parseInt(d.impressions || 0),
          clicks:      parseInt(d.clicks      || 0),
          ctr:         parseFloat(d.ctr       || 0),
          cpm:         parseFloat(d.cpm       || 0),
          cpc:         parseFloat(d.cpc       || 0),
          spend:       parseFloat(d.spend     || 0)
        };
      } else {
        result[as.key] = { name: as.name, noData: true };
      }
    } catch(e) {
      result[as.key] = { name: as.name, error: e.message };
    }
  });

  return result;
}

//  3. 예산 결정 및 Meta API 적용
// ════════════════════════════════════════════════

function applyWeeklyBudget(counts) {
  // 날짜별 여성 부족분 계산 → 가장 급한 날 기준
  var maxShort = 0;
  var keys = Object.keys(counts);
  for (var i = 0; i < keys.length; i++) {
    var short = Math.max(0, FEMALE_GOAL - counts[keys[i]].female);
    if (short > maxShort) maxShort = short;
  }

  // 예산표에서 해당 구간 찾기
  var budget1 = 10000;
  for (var j = 0; j < BUDGET_TABLE.length; j++) {
    var row = BUDGET_TABLE[j];
    if (maxShort >= row[0] && maxShort <= row[1]) {
      budget1 = row[2];
      break;
    }
  }

  // Meta API 예산 변경
  var ok1 = setAdsetDailyBudget(ADSET_FEMALE_MAIN, budget1);

  return {
    maxShort:     maxShort,
    budget1:      budget1,
    ok1:          ok1,
    manualNeeded: maxShort >= 15
  };
}

function setAdsetDailyBudget(adsetId, budget) {
  // 범위 벗어난 값이면 실행 차단 + 즉시 알림
  if (typeof budget !== 'number' || budget < MIN_DAILY_BUDGET || budget > MAX_DAILY_BUDGET) {
    var msg = '[예산 오류] 비정상 값 차단: adsetId=' + adsetId + ', budget=' + budget
            + ' (허용 범위: ₩' + MIN_DAILY_BUDGET.toLocaleString()
            + ' ~ ₩' + MAX_DAILY_BUDGET.toLocaleString() + ')';
    GmailApp.sendEmail(NOTIFY_EMAIL, '[썸잇] 🚨 예산 오류 — 자동 차단됨', msg);
    return false;
  }
  try {
    var res  = UrlFetchApp.fetch('https://graph.facebook.com/v21.0/' + adsetId, {
      method: 'post',
      contentType: 'application/x-www-form-urlencoded',
      payload: 'daily_budget=' + budget + '&access_token=' + ADS_ACCESS_TOKEN,
      muteHttpExceptions: true
    });
    var json = JSON.parse(res.getContentText());
    return json.success === true;
  } catch(e) {
    return false;
  }
}


// ════════════════════════════════════════════════
//  4. 소재 피로도 체크
// ════════════════════════════════════════════════

function checkCreativeFatigue(adPerf) {
  var alerts = [];
  var keys = Object.keys(adPerf);
  for (var i = 0; i < keys.length; i++) {
    var p = adPerf[keys[i]];
    if (p.error || p.noData) continue;
    // CTR 1% 미만 + 노출 3,000회 이상 → 소재 교체 권고
    if (p.ctr < 1.0 && p.impressions >= 3000) {
      alerts.push('⚠️ [소재 교체 필요] ' + p.name
        + ' — CTR ' + p.ctr.toFixed(2) + '%'
        + ' (노출 ' + p.impressions.toLocaleString() + '회)');
    }
  }
  return alerts;
}

function sendWeeklyReport(counts, adPerf, budgetResult, alerts) {

  var dateKeys = Object.keys(counts);

  lines.push('[썸잇] 주간 광고 자동 보고서');
  lines.push('');

  // 파티 현황
  lines.push('■ 다음 주 파티 현황');
  for (var i = 0; i < dateKeys.length; i++) {
    var c = counts[dateKeys[i]];
    var fs = Math.max(0, FEMALE_GOAL - c.female);
    var ms = Math.max(0, MALE_GOAL   - c.male);
    lines.push('  ' + c.label
      + '  남' + c.male   + '명 (부족 ' + ms + ')'
      + '  여' + c.female + '명 (부족 ' + fs + ')');
  }
  lines.push('');

  // 예산 조정 결과
  lines.push('■ 예산 자동 조정 (여성 최대 부족: ' + budgetResult.maxShort + '명)');
  lines.push('  여성_19-34_부산_잠재고객 → ₩' + budgetResult.budget1.toLocaleString() + '/일  ' + (budgetResult.ok1 ? '✅ 적용' : '❌ 실패'));
  lines.push('  남성_부산_잠재고객 → 변경 없음');
  lines.push('');

  // 수동 모집 알림
  if (budgetResult.manualNeeded) {
    lines.push('[주의] 수동 모집 필요');
    lines.push('  여성 ' + budgetResult.maxShort + '명 부족 — 광고만으로 달성 어렵습니다');
    lines.push('  SNS / 지인 모집 병행을 권고합니다');
    lines.push('');
  }

  // 광고 성과
  lines.push('■ 광고 성과 (최근 7일)');
  var perfKeys = Object.keys(adPerf);
  for (var j = 0; j < perfKeys.length; j++) {
    var p = adPerf[perfKeys[j]];
    if (p.error) {
      lines.push('  ' + p.name + ': 조회 실패');
    } else if (p.noData) {
      lines.push('  ' + p.name + ': 데이터 없음');
    } else {
      lines.push('  ' + p.name);
      lines.push('    CTR ' + p.ctr.toFixed(2) + '%'
        + ' | CPM ₩' + Math.round(p.cpm).toLocaleString()
        + ' | CPC ₩' + Math.round(p.cpc).toLocaleString()
        + ' | 집행 ₩' + Math.round(p.spend).toLocaleString());
    }
  }
  lines.push('');

  // 소재 알림
  if (alerts.length > 0) {
    lines.push('■ 소재 알림');
    for (var k = 0; k < alerts.length; k++) {
      lines.push('  ' + alerts[k]);
    }
  } else {
    lines.push('■ 소재: 교체 불필요 (CTR 양호)');
  }

  var subject = '[썸잇] 주간 광고 보고서'
    + ' — 여성 최대부족 ' + budgetResult.maxShort + '명'
    + (budgetResult.manualNeeded ? ' 🚨 수동모집 필요' : '');

  GmailApp.sendEmail(NOTIFY_EMAIL, subject, lines.join('\n'));

//  일일 광고 보고서 이메일 발송 (CCR 자동화용)
// ════════════════════════════════════════════════

function handleAdReport(data) {
  if ((data.key || '') !== ADMIN_KEY)
    return jsonResponse({ result: 'error', message: '인증실패' });
  const subject = '[썸잇] 일일 광고 체킹 — '
    + Utilities.formatDate(new Date(), 'Asia/Seoul', 'M/d(E)');
  GmailApp.sendEmail('aprycotkr@naver.com', subject, data.report || '');
  return jsonResponse({ result: 'success' });
}
